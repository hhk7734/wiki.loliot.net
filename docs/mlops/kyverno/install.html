Kyverno for Kubernetes Policy Management

설치​

  • Kyverno 설치 방법
helm repo add kyverno https://kyverno.github.io/kyverno/
helm repo update kyverno \
&& helm search repo kyverno/kyverno -l | head -n 10
helm pull kyverno/kyverno --version 3.2.6
helm show values kyverno/kyverno --version 3.2.6 > kyverno-3.2.6.yaml
helm template kyverno kyverno-3.2.6.tgz \
    -n kyverno \
    -f kyverno-values.yaml \
    > kyverno.yaml
helm upgrade -i kyverno kyverno-3.2.6.tgz \
    --history-max 5 \
    -n kyverno \
    -f kyverno-values.yaml
